def solution(clothes):
    check = {}

    for i in clothes:
        check[i[1]] = 0
    
    for i in clothes:
        check[i[1]] += 1
    
    if len(check)>1:
        sum = 1
        for i in check :
            sum = sum * check[i]
        answer = len(clothes) + sum
    else : answer = len(clothes)
    
    return answer
